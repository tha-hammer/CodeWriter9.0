---- MODULE CrawlStoreHosted ----
EXTENDS Integers, FiniteSets, Sequences, TLC

CONSTANTS MaxOps

ASSUME MaxOps \in Nat /\ MaxOps > 0

(* --algorithm CrawlStoreHosted

variables
    mode \in {"hosted", "local"},
    backend = "none",
    conn_state = "closed",
    context_entered = FALSE,
    context_exited = FALSE,
    current_op = "idle",
    op_result = "none",
    step_count = 0,
    isolation_mode = "none",
    ops_performed = 0,
    do_exit = FALSE;

define

    PublicInterface ==
        { "get_record", "upsert_record", "insert_record",
          "get_all_uuids", "get_pending_uuids",
          "get_stale_records", "get_card_text" }

    BackendValid ==
        backend \in {"none", "postgres", "sqlite"}

    HostedUsesPostgres ==
        (mode = "hosted" /\ context_entered) => backend = "postgres"

    LocalUsesSQLite ==
        (mode = "local" /\ context_entered) => backend = "sqlite"

    BackendDeterminedByMode ==
        context_entered =>
            (mode = "hosted" <=> backend = "postgres")

    ConnNeverOpenBeforeEnter ==
        ~context_entered => conn_state = "closed"

    ConnOpenDuringContext ==
        (context_entered /\ ~context_exited) => conn_state = "open"

    ConnClosedAfterExit ==
        context_exited => conn_state = "closed"

    IsolationSetOnEnter ==
        (context_entered /\ ~context_exited) => isolation_mode = "wal_equivalent"

    OperationsOnlyWhileConnected ==
        ops_performed > 0 => context_entered

    CallerNeedsNoModification ==
        current_op \in PublicInterface \union {"idle"}

    BoundedExecution ==
        step_count <= MaxOps

end define;

fair process crawlstore = "worker"
begin
    Init:
        if mode = "hosted" then
            backend := "postgres";
        else
            backend := "sqlite";
        end if;
        step_count := step_count + 1;
    Enter:
        conn_state := "open";
        isolation_mode := "wal_equivalent";
        context_entered := TRUE;
        step_count := step_count + 1;
    Operate:
        while step_count < MaxOps /\ ~do_exit do
            either
                current_op := "get_record";
            or
                current_op := "upsert_record";
            or
                current_op := "insert_record";
            or
                current_op := "get_all_uuids";
            or
                current_op := "get_pending_uuids";
            or
                current_op := "get_stale_records";
            or
                current_op := "get_card_text";
            or
                do_exit := TRUE;
            end either;
        FinishOp:
            op_result     := IF ~do_exit THEN "result"          ELSE op_result;
            ops_performed := IF ~do_exit THEN ops_performed + 1 ELSE ops_performed;
            step_count    := IF ~do_exit THEN step_count + 1    ELSE step_count;
            current_op    := IF ~do_exit THEN "idle"            ELSE current_op;
        end while;
    ExitCtx:
        conn_state     := "closed";
        context_exited := TRUE;
        step_count     := step_count + 1;
    Terminate:
        skip;
end process;

end algorithm; *)

AllInvariants ==
    /\ BackendValid
    /\ HostedUsesPostgres
    /\ LocalUsesSQLite
    /\ BackendDeterminedByMode
    /\ ConnNeverOpenBeforeEnter
    /\ ConnOpenDuringContext
    /\ ConnClosedAfterExit
    /\ IsolationSetOnEnter
    /\ OperationsOnlyWhileConnected
    /\ CallerNeedsNoModification
    /\ BoundedExecution

EventuallyTerminates ==
    <>(context_exited /\ conn_state = "closed")
