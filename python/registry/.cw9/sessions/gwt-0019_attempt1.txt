MODULE ProjectProvision

EXTENDS Integers, FiniteSets, Sequences, TLC

CONSTANTS
    MaxRequests,
    MaxProjectIds

ASSUME MaxRequests \in Nat /\ MaxRequests >= 1
ASSUME MaxProjectIds \in Nat /\ MaxProjectIds >= MaxRequests

    storage_prefixes = {},
