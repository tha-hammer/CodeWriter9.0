------------------------ MODULE GWT0026PromptContext ------------------------

EXTENDS Integers, FiniteSets, Sequences, TLC

CONSTANTS
    GWT_ID,
    MaxSteps

ASSUME MaxSteps \in Nat /\ MaxSteps > 0

(*
 * States for the query_context / _build_prompt pipeline.
 *
 * Idle            -> worker is ready, workspace files may or may not include crawl.db
 * QueryingDag     -> dag.query_relevant(gwt_id) is in flight
 * FetchingRecords -> CrawlStore.get_record() calls are in flight (crawl.db present)
 * FormattingCtx   -> format_prompt_context() is in flight
 * BuildingPrompt  -> _build_prompt() is in flight
 * PromptReady     -> terminal: prompt successfully built
 *)

VARIABLES
    current_state,
    crawl_db_present,
    dag_nodes,
    fn_records,
    combined_context,
    prompt,
    error_occurred,
    step_count

StateSet == {
    "Idle",
    "QueryingDag",
    "FetchingRecords",
    "FormattingCtx",
    "BuildingPrompt",
    "PromptReady"
}

TerminalStates == {"PromptReady"}

vars == <<
    current_state,
    crawl_db_present,
    dag_nodes,
    fn_records,
    combined_context,
    prompt,
    error_occurred,
    step_count
>>

TypeInvariant ==
    /\ current_state   \in StateSet
    /\ crawl_db_present \in BOOLEAN
    /\ dag_nodes       \in BOOLEAN
    /\ fn_records      \in BOOLEAN
    /\ combined_context \in BOOLEAN
    /\ prompt          \in BOOLEAN
    /\ error_occurred  \in BOOLEAN
    /\ step_count      \in Nat

PromptBuiltImpliesContextFormatted ==
    prompt => combined_context

ContextFormattedImpliesDAGQueried ==
    combined_context => dag_nodes

NoCrawlDbMeansNoFnRecords ==
    ~crawl_db_present => ~fn_records

NoCrawlDbNoError ==
    ~crawl_db_present => ~error_occurred

PromptAlwaysEventuallyBuilt ==
    <>(current_state = "PromptReady")

BoundedExecution ==
    step_count <= MaxSteps

Invariants ==
    /\ TypeInvariant
    /\ PromptBuiltImpliesContextFormatted
    /\ ContextFormattedImpliesDAGQueried
    /\ NoCrawlDbMeansNoFnRecords
    /\ NoCrawlDbNoError
    /\ BoundedExecution
