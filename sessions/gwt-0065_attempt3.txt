---- MODULE SeamSatisfiedNoReport ----

EXTENDS Integers, FiniteSets, TLC

CONSTANTS
    TypeValues,
    CompatPairs,
    CallerType,
    CalleeType

ASSUME CompatPairs \subseteq (TypeValues \X TypeValues)
ASSUME \A t \in TypeValues : <<t, t>> \in CompatPairs
ASSUME CallerType \in TypeValues
ASSUME CalleeType \in TypeValues
ASSUME <<CallerType, CalleeType>> \in CompatPairs

(*--algorithm SeamSatisfiedNoReport

variables
    provided_type = CallerType,
    expected_type = CalleeType,
    compatible = FALSE,
    result = {},
    phase = "init";

define

    NoFalsePositive ==
        (phase = "done" /\ <<provided_type, expected_type>> \in CompatPairs)
            => result = {}

    ResultEmptyWhenCompatible ==
        (phase = "done" /\ compatible = TRUE) => result = {}

    SeamSatisfied ==
        phase = "done" => result = {}

    PhaseValid ==
        phase \in {"init", "loading", "checking", "confirming", "done"}

end define;

fair process checker = "seam_checker"
begin
    Start:
        phase := "loading";

    CheckSeam:
        if <<provided_type, expected_type>> \in CompatPairs then
            compatible := TRUE;
            phase := "confirming";
        else
            compatible := FALSE;
            phase := "confirming";
        end if;

    ConfirmCompatible:
        if compatible = TRUE then
            result := {};
            phase := "done";
        else
            result := { [ provided_type |-> provided_type,
                          expected_type |-> expected_type,
                          severity      |-> "type_mismatch" ] };
            phase := "done";
        end if;

    Finish:
        assert result = {};
        assert compatible = TRUE;

end process;

end algorithm; *)
