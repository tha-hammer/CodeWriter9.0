class TestEdgeCases:

    def test_isolated_node_subgraph(self) -> None:
        """Isolated node: no edges; visited={A}, frontier={} immediately.

        FIX (was B1): the original called _assert_all_invariants which invokes
        _assert_chain_fully_covered and _assert_completeness_invariant.  Both
        helpers require REACHABLE = {NA, NB, NC} ⊆ visited, but visited = {NA}

        visited = _node_ids(result)

        _assert_seed_always_included(visited, "complete")

        with pytest.raises(NodeNotFoundError):
            dag.extract_subgraph(NA)

    def test_diamond_pattern_all_reachable(self) -> None:
        """
        Diamond: A→B, A→C, B→D, C→D.
        Reachable from A = {A, B, C, D}.  All invariants must hold.

        FIX (was W2): the original used len(visited)-1 = 3 as the depth proxy.
        For a diamond the maximum BFS depth is 2 (A→B→D or A→C→D), not 3.
        Use the correct BFS depth literal.
        """
        ND = "D"
        dag = RegistryDag()
        for nid in (NA, NB, NC, ND):
            dag.add_node(Node.behavior(nid, nid, "g", "w", "t"))
        dag.add_edge(Edge(NA, NB, EdgeType.IMPORTS))
        dag.add_edge(Edge(NA, NC, EdgeType.IMPORTS))
        dag.add_edge(Edge(NB, ND, EdgeType.IMPORTS))
        dag.add_edge(Edge(NC, ND, EdgeType.IMPORTS))

        visited = _node_ids(result)

        assert {NA, NB, NC, ND}.issubset(visited)
        _assert_seed_always_included(visited, "complete")
        _assert_valid_phase("complete")
        _assert_bounded_depth(2)   # BFS depth = 2 (A→{B,C}→D); well within MaxSteps=4
        _assert_frontier_subset_visited(set(), visited)

    def test_seed_from_middle_of_chain(self) -> None:

        Subgraph from A must not include D or E.

    def test_wide_fan_out_bounded_depth(self) -> None:
        """
        A node with four direct successors: A→{B,C,D,E}.
        All must be reachable at depth=1, well within MaxSteps=4.
        """
        children = [NB, NC, "D", "E"]
        dag = RegistryDag()
        dag.add_node(Node.behavior(NA, NA, "g", "w", "t"))

            dag.add_edge(Edge(NA, child, EdgeType.IMPORTS))

        result = dag.extract_subgraph(NA)
        visited = _node_ids(result)

        expected = {NA} | set(children)
        assert expected.issubset(visited)
        _assert_bounded_depth(1)
        _assert_frontier_subset_visited(set(), visited)

    def test_cycle_edge_rejected_or_traversal_still_terminates(self) -> None:
        """
        Self-loops / back-edges: either CycleError is raised at add_edge time
        (cycle detection), OR if accepted, extract_subgraph must still terminate.
        In either case BoundedDepth and ValidPhase must not be violated.

        FIX (was B2): the original called _assert_bounded_depth(MAX_STEPS),

            result = dag.extract_subgraph(NA)
            visited = _node_ids(result)
            assert NA in visited
            _assert_valid_phase("complete")
            # Use the actual node count as the depth proxy; for a self-loop on
            # a single node the traversal visits {A} only, so depth ≤ 0.
            _assert_bounded_depth(len(visited) - 1)
