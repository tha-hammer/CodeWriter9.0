--------------------------- MODULE GracefulShutdown ---------------------------

EXTENDS Integers, FiniteSets, TLC

CONSTANTS
    Tasks

ASSUME Tasks # {} /\ IsFiniteSet(Tasks)

(*--algorithm GracefulShutdown

variables
    shutdown_requested   = FALSE,
    in_flight            = {},
    completed            = {},
    started_after_signal = {},
    in_flight_at_signal  = {},
    pending              = Tasks,
    run_active           = TRUE;

define

    TypeOK ==
        /\ shutdown_requested   \in BOOLEAN
        /\ in_flight            \subseteq Tasks
        /\ completed            \subseteq Tasks
        /\ started_after_signal \subseteq Tasks
        /\ in_flight_at_signal  \subseteq Tasks
        /\ pending              \subseteq Tasks
        /\ run_active           \in BOOLEAN
        /\ phase                \in {"running", "draining"}
        /\ candidate            \in Tasks \union {"none"}

    \* SAFETY: No extraction is started after the shutdown flag is set
    NoNewAfterShutdown ==
        shutdown_requested => started_after_signal = {}

    \* SAFETY: Once the signal fires, in_flight can only shrink — no new entries
    InFlightDoesNotGrow ==
        shutdown_requested => in_flight \subseteq in_flight_at_signal

    \* SAFETY: run() does not return until every in-flight extraction has completed
    InFlightCompletes ==
        ~run_active => in_flight = {}

    \* SAFETY: Completed set is always a subset of known tasks
    CompletedSubsetTasks ==
        completed \subseteq Tasks

    \* STRUCTURE: pending, in_flight, and completed are pairwise disjoint
    Partitioned ==
        /\ (in_flight \intersect completed) = {}
        /\ (pending   \intersect completed) = {}
        /\ (pending   \intersect in_flight) = {}

    \* MONOTONICITY: shutdown_requested is a one-way latch (enforced structurally
    \* by signal_handler, which only ever sets it — never clears it)
    ShutdownMonotonicInv ==
        \* Captured structurally: signal_handler sets once, never unsets
        TRUE

end define;

\* Models the OS delivering SIGINT; fires at most once, nondeterministically
fair process signal_handler = "signal"
begin
    FireSignal:
        either
            shutdown_requested  := TRUE;
            in_flight_at_signal := in_flight;
        or
            skip;
        end either;
end process;

\* Models CrawlOrchestrator.run(): guards each extraction start against the flag,
\* then drains all in-flight work before marking itself done
fair process orchestrator = "orchestrator"
variables
    phase     = "running",
    candidate = "none";
begin
    OrchestratorLoop:
        while phase = "running" do
            either
                \* Pre-check: only proceed when shutdown has not been requested
                await ~shutdown_requested /\ pending # {};
                with t \in pending do
                    candidate := t;
                    pending   := pending \ {t};
                end with;
            StartOrAbort:
                \* Post-check: a signal may have arrived between the two steps;
                \* if so, return the candidate to pending rather than starting it
                if ~shutdown_requested then
                    in_flight := in_flight \union {candidate};
                else
                    pending := pending \union {candidate};
                end if;
                candidate := "none";
            or
                \* Cooperative exit: shutdown received or no work remains
                await shutdown_requested \/ pending = {};
                phase := "draining";
            end either;
        end while;
    DrainInflight:
        \* Block until every extraction that was already in-flight has finished
        await in_flight = {};
    OrchestratorDone:
        run_active := FALSE;
end process;

\* Models the extract_one workers: complete in-flight calls and persist results
fair process completer = "completer"
begin
    WaitForWork:
        await in_flight # {} \/ ~run_active;
    CheckWork:
        if in_flight # {} then
            with t \in in_flight do
                in_flight := in_flight \ {t};
                completed := completed \union {t};
            end with;
            goto WaitForWork;
        end if;
    CompleterFinish:
        skip;
end process;

end algorithm; *)
