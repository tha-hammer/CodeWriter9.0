    phase         = "skeleton",
    entry_points  = {},
    rem_skels     = 1..Skeletons,
    rem_files     = 1..SourceFiles,
    cur_file      = 0,
    cur_patterns  = {},
    step_count    = 0;

define

    AllEntryTypesValid ==

            ep.entry_type \in {MAIN_TYPE, PUBLIC_API_TYPE,
                               HTTP_ROUTE_TYPE, CLI_COMMAND_TYPE}

    MainIsTopLevel ==
        \A ep \in entry_points :
            ep.entry_type = MAIN_TYPE => ep.function_name = "main"

    PublicOnlyForLibrary ==
        \A ep \in entry_points :
            ep.entry_type = PUBLIC_API_TYPE => CodebaseType = "library"

    HandleFuncMethodAgnostic ==
        \A ep \in entry_points :
            (ep.entry_type = HTTP_ROUTE_TYPE /\ ep.pat_type = "handlefunc") =>
                ep.method = NoMethod

    TestFilesNotScanned ==
        \A ep \in entry_points :
            ep.entry_type \in {HTTP_ROUTE_TYPE, CLI_COMMAND_TYPE} =>
                FileIsTest[ep.file_path] = FALSE

    DotFilesNotScanned ==
        \A ep \in entry_points :
            ep.entry_type \in {HTTP_ROUTE_TYPE, CLI_COMMAND_TYPE} =>
                FileIsDot[ep.file_path] = FALSE

        ~(\E ep1, ep2 \in entry_points :
            ep1.file_path     = ep2.file_path     /\
            ep1.function_name = ep2.function_name  /\
            ep1.entry_type    = MAIN_TYPE          /\
            ep2.entry_type    = PUBLIC_API_TYPE)

    DeduplicatedResults ==
        \A ep1, ep2 \in entry_points :
            ( ep1.file_path     = ep2.file_path     /\
              ep1.function_name = ep2.function_name  /\
              ep1.entry_type    = ep2.entry_type     /\
              ep1.route         = ep2.route          /\
              ep1.method        = ep2.method ) => ep1 = ep2

    BoundedExecution == step_count <= MaxSteps

    Completion ==
        phase = "complete" =>
            rem_skels = {} /\ rem_files = {} /\ cur_patterns = {}

end define;

                            SkelVisibility[idx] = "public"  /\
                            CodebaseType        = "library" /\
                            SkelFuncName[idx]  /= "main"

                                 function_name |-> SkelFuncName[idx],
                                 entry_type    |-> PUBLIC_API_TYPE,
                                 route         |-> NoRoute,
                                 method        |-> NoMethod,
                                 pat_type      |-> "skeleton" ]}
                         ELSE entry_points;
            end with;
        end while;

    TransitionToScan:
        phase := "scanning";

    FileLoop:
        while rem_files /= {} /\ step_count < MaxSteps do

                rem_files    := rem_files \ {fidx}  ||
                step_count   := step_count + 1      ||
                cur_file     := fidx                ||
                cur_patterns :=
                    IF FileIsTest[fidx] = FALSE /\ FileIsDot[fidx] = FALSE
                    THEN FilePats[fidx]
                    ELSE {};
            end with;
            PatternInnerLoop:
                while cur_patterns /= {} do

                        cur_patterns := cur_patterns \ {pat} ||

                               pat.type = "echo_route" \/
                               pat.type = "chi_route"

                                    file_path     |-> cur_file,
                                    function_name |-> pat.handler,
                                    entry_type    |-> HTTP_ROUTE_TYPE,
                                    route         |-> pat.route,
                                    method        |-> pat.method,
                                    pat_type      |-> pat.type ]}
                            ELSE IF pat.type = "handlefunc"
                                 THEN entry_points \cup {[
                                         file_path     |-> cur_file,
                                         function_name |-> pat.handler,
                                         entry_type    |-> HTTP_ROUTE_TYPE,

                                         method        |-> NoMethod,
                                         pat_type      |-> "handlefunc" ]}

                                               file_path     |-> cur_file,
                                               function_name |-> pat.use,
                                               entry_type    |-> CLI_COMMAND_TYPE,
                                               route         |-> NoRoute,
                                               method        |-> NoMethod,
                                               pat_type      |-> "cobra" ]}
                                      ELSE entry_points;
                    end with;
                end while;
        end while;

    Finish:
        phase := "complete";

end process;
