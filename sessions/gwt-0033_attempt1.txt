variables
  loop_result = "idle",
  compiled_spec_exists = TRUE,

  spec_dir_has_traces = FALSE,
  phase = "start",
  step_count = 0;

define
  PassImpliesWrite ==

  FailImpliesNoWrite ==

end define;
