---- MODULE ForwardSubgraphComplete ----

EXTENDS Integers, FiniteSets, TLC

CONSTANTS
    Nodes,
    StartNode,
    Edges,
    MaxDepth

ASSUME StartNode \in Nodes
ASSUME \A e \in Edges : e[1] \in Nodes /\ e[2] \in Nodes

(* --algorithm ForwardSubgraphComplete

variables
    visited  = {},
    frontier = {},
    phase    = "init",
    depth    = 0;

define

    Reachable ==
        LET RECURSIVE Reach(_, _)
            Reach(S, boundary) ==
                IF boundary = {} THEN S
                ELSE
                    LET next == { e[2] : e \in { ed \in Edges :
                                                  ed[1] \in boundary /\
                                                  ed[2] \in Nodes } } \ S
                    IN  Reach(S \cup next, next)
        IN Reach({StartNode}, {StartNode})

    SeedAlwaysIncluded ==
        phase /= "init" => StartNode \in visited

    VisitedSubsetReachable ==
        visited \subseteq Reachable

    CompletenessInvariant ==
        phase = "complete" => Reachable \subseteq visited

    FrontierSubsetVisited ==
        frontier \subseteq visited

    BoundedDepth ==
        depth <= MaxDepth

    ValidPhase ==
        phase \in {"init", "expanding", "complete"}

end define;

fair process traversal = "traversal"
begin
    Seed:
        visited  := {StartNode};
        frontier := {StartNode};
        phase    := "expanding";
        depth    := 0;

    Expand:
        while frontier /= {} /\ depth < MaxDepth do
            with node \in frontier do
                with newCallees = { e[2] : e \in { ed \in Edges :
                                                    ed[1] = node /\
                                                    ed[2] \in Nodes } } \ visited
                do
                    visited  := visited \cup newCallees;
                    frontier := (frontier \ {node}) \cup newCallees;
                    depth    := depth + 1;
                end with;
            end with;
        end while;

    Terminate:
        phase := "complete";

end process;

end algorithm; *)
