EXTENDS Integers, FiniteSets, TLC

CONSTANTS
    Nodes,
    Hashes,
    MaxSteps

ASSUME MaxSteps \in Nat /\ MaxSteps > 0
ASSUME Nodes /= {} /\ Hashes /= {}
