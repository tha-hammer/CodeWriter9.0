## Review Findings

Two semantic issues were found:

### Issue 1 — Unused import: `NodeKind`

`NodeKind` is imported from `registry.types` but is never referenced anywhere in the file. This is a dead import that can cause linter failures and signals an incomplete abstraction.

**Fix:** Remove `NodeKind` from the import line.

---

### Issue 2 — `test_intermediate_state_after_determine_mode` never calls the SUT

The test claims to verify "State 2 assertions: mode and reviews_list are set, determined still False," but it **never calls `determine_mode()`**. It only computes `mode_intermediate` and `reviews_intermediate` from raw local Python arithmetic and asserts on those — which are trivially and vacuously true regardless of the SUT's behavior. No `determined` variable is ever checked in the test either, despite the docstring promising that assertion.

Because `determine_mode()` is an atomic pure function (DetermineMode + Finish are not separately callable), intermediate state is not directly observable. The test should call the SUT and verify the final committed state, with a note explaining why intermediate state is unobservable.

**Fix:** Rewrite the test to call `determine_mode()` and assert on all three return values, with corrected documentation.

---

All other elements are semantically correct:
- Every TLA+ invariant in `_assert_all_invariants()` precisely matches its formal specification
- The conflict branch (`self_flag=True AND external_flag=True`) correctly leaves `determined=False` and all invariant guards are vacuously satisfied
- Fixture topologies cover the required breadth (empty DAG, singleton, diamond, multi-node)
- The `dag_trace*` fixtures correctly encode their trace initial conditions
- The exhaustive 2³ coverage test properly closes the invariant suite
- `CycleError` and `NodeNotFoundError` edge cases exercise structural DAG guarantees independently of mode logic

def _make_dag(
    nodes: list[str],
    edges: list[tuple[str, str]],
    artifacts: dict | None = None,
) -> RegistryDag:
    """Build a RegistryDag from trace-derived topology data."""
    dag = RegistryDag()
    for nid in nodes:
        dag.add_node(Node.behavior(nid, nid, given="g", when="w", then="t"))
    for src, dst in edges:
        dag.add_edge(Edge(src, dst, EdgeType.IMPORTS))
    dag.test_artifacts = artifacts if artifacts is not None else {}
    return dag


def determine_mode(
    self_flag: bool,
    external_flag: bool,
    cw9_dir_exists: bool,
) -> tuple[str, str, bool]:
    """
    Python translation of the TLA+ PlusCal DetermineMode / Finish algorithm.

    Returns
    -------
    (mode, reviews_list, determined)
      mode        – "self" | "external" | "unset"
      reviews_list – "SELF_REVIEWS" | "EXTERNAL_REVIEWS" | "UNSET"
      determined  – True when the algorithm committed to a mode
    """
    mode = "unset"
    reviews_list = "UNSET"
    determined = False

    # DetermineMode step: skip if BOTH flags are set simultaneously (conflict)
    if not (self_flag and external_flag):
        if self_flag:
            mode = "self"
            reviews_list = "SELF_REVIEWS"
        elif external_flag:
            mode = "external"
            reviews_list = "EXTERNAL_REVIEWS"
        elif cw9_dir_exists:
            mode = "external"
            reviews_list = "EXTERNAL_REVIEWS"
        else:
            mode = "self"
            reviews_list = "SELF_REVIEWS"

    # Finish step
    if not (self_flag and external_flag):
        determined = True

    return mode, reviews_list, determined


def _assert_all_invariants(
    mode: str,
    reviews_list: str,
    determined: bool,
    self_flag: bool,
    external_flag: bool,
    cw9_dir_exists: bool,
    *,
    label: str = "",
) -> None:
    """Assert every TLA+ invariant for the given state variables."""
    ctx = f" [{label}]" if label else ""

    # ModeValid
    if determined:
        assert mode in {"self", "external"}, (
            f"ModeValid violated{ctx}: determined=True but mode={mode!r}"
        )

    # ReviewsValid
    if determined:
        assert reviews_list in {"SELF_REVIEWS", "EXTERNAL_REVIEWS"}, (
            f"ReviewsValid violated{ctx}: determined=True but reviews_list={reviews_list!r}"
        )

    # ModeReviewConsistency
    if determined:
        if mode == "self":
            assert reviews_list == "SELF_REVIEWS", (
                f"ModeReviewConsistency violated{ctx}: mode=self but reviews_list={reviews_list!r}"
            )
        if mode == "external":
            assert reviews_list == "EXTERNAL_REVIEWS", (
                f"ModeReviewConsistency violated{ctx}: mode=external but reviews_list={reviews_list!r}"
            )

    # ExplicitOverrideSelf
    if determined and self_flag and not external_flag:
        assert mode == "self", (
            f"ExplicitOverrideSelf violated{ctx}: self_flag=True, external_flag=False, "
            f"determined=True but mode={mode!r}"
        )

    # ExplicitOverrideExternal
    if determined and external_flag and not self_flag:
        assert mode == "external", (
            f"ExplicitOverrideExternal violated{ctx}: external_flag=True, self_flag=False, "
            f"determined=True but mode={mode!r}"
        )

    # AutoDetectExternal
    if determined and not self_flag and not external_flag and cw9_dir_exists:
        assert mode == "external", (
            f"AutoDetectExternal violated{ctx}: no flags, cw9_dir_exists=True, "
            f"determined=True but mode={mode!r}"
        )

    # AutoDetectSelf
    if determined and not self_flag and not external_flag and not cw9_dir_exists:
        assert mode == "self", (
            f"AutoDetectSelf violated{ctx}: no flags, cw9_dir_exists=False, "
            f"determined=True but mode={mode!r}"
        )

### Issue 2 — `test_intermediate_state_after_determine_mode` never calls the SUT

The test claims to verify "State 2 assertions: mode and reviews_list are set, determined still False," but it **never calls `determine_mode()`**. It only computes `mode_intermediate` and `reviews_intermediate` from raw local Python arithmetic and asserts on those — which are trivially and vacuously true regardless of the SUT's behavior. No `determined` variable is ever checked in the test either, despite the docstring promising that assertion.

Because `determine_mode()` is an atomic pure function (DetermineMode + Finish are not separately callable), intermediate state is not directly observable. The test should call the SUT and verify the final committed state, with a note explaining why intermediate state is unobservable.

**Fix:** Rewrite the test to call `determine_mode()` and assert on all three return values, with corrected documentation.

TLA+ PlusCal algorithm; it is the system-under-test for every assertion.

  ReviewsValid         – determined => reviews_list ∈ {"SELF_REVIEWS","EXTERNAL_REVIEWS"}

  AutoDetectExternal   – (determined ∧ ¬flags ∧ cw9_dir_exists) => mode="external"
  AutoDetectSelf       – (determined ∧ ¬flags ∧ ¬cw9_dir_exists) => mode="self"

def _make_dag(
    nodes: list[str],
    edges: list[tuple[str, str]],
    artifacts: dict | None = None,
) -> RegistryDag:
    """Build a RegistryDag from trace-derived topology data."""
    dag = RegistryDag()
    for nid in nodes:
        dag.add_node(Node.behavior(nid, nid, given="g", when="w", then="t"))
    for src, dst in edges:
        dag.add_edge(Edge(src, dst, EdgeType.IMPORTS))
    dag.test_artifacts = artifacts if artifacts is not None else {}
    return dag


def determine_mode(
    self_flag: bool,
    external_flag: bool,
    cw9_dir_exists: bool,
) -> tuple[str, str, bool]:
    """
    Python translation of the TLA+ PlusCal DetermineMode / Finish algorithm.

    Returns
    -------
    (mode, reviews_list, determined)
      mode        – "self" | "external" | "unset"
      reviews_list – "SELF_REVIEWS" | "EXTERNAL_REVIEWS" | "UNSET"
      determined  – True when the algorithm committed to a mode
    """
    mode = "unset"
    reviews_list = "UNSET"
    determined = False

    # DetermineMode step: skip if BOTH flags are set simultaneously (conflict)
    if not (self_flag and external_flag):
        if self_flag:
            mode = "self"
            reviews_list = "SELF_REVIEWS"
        elif external_flag:
            mode = "external"
            reviews_list = "EXTERNAL_REVIEWS"
        elif cw9_dir_exists:
            mode = "external"
            reviews_list = "EXTERNAL_REVIEWS"
        else:
            mode = "self"
            reviews_list = "SELF_REVIEWS"

    # Finish step
    if not (self_flag and external_flag):
        determined = True

    return mode, reviews_list, determined


def _assert_all_invariants(
    mode: str,
    reviews_list: str,
    determined: bool,
    self_flag: bool,
    external_flag: bool,
    cw9_dir_exists: bool,
    *,
    label: str = "",
) -> None:
    """Assert every TLA+ invariant for the given state variables."""
    ctx = f" [{label}]" if label else ""

    # ModeValid
    if determined:
        assert mode in {"self", "external"}, (
            f"ModeValid violated{ctx}: determined=True but mode={mode!r}"
        )

    # ReviewsValid
    if determined:
        assert reviews_list in {"SELF_REVIEWS", "EXTERNAL_REVIEWS"}, (
            f"ReviewsValid violated{ctx}: determined=True but reviews_list={reviews_list!r}"
        )

    # ModeReviewConsistency
    if determined:
        if mode == "self":
            assert reviews_list == "SELF_REVIEWS", (
                f"ModeReviewConsistency violated{ctx}: mode=self but reviews_list={reviews_list!r}"
            )
        if mode == "external":
            assert reviews_list == "EXTERNAL_REVIEWS", (
                f"ModeReviewConsistency violated{ctx}: mode=external but reviews_list={reviews_list!r}"
            )

    # ExplicitOverrideSelf
    if determined and self_flag and not external_flag:
        assert mode == "self", (
            f"ExplicitOverrideSelf violated{ctx}: self_flag=True, external_flag=False, "
            f"determined=True but mode={mode!r}"
        )

    # ExplicitOverrideExternal
    if determined and external_flag and not self_flag:
        assert mode == "external", (
            f"ExplicitOverrideExternal violated{ctx}: external_flag=True, self_flag=False, "
            f"determined=True but mode={mode!r}"
        )

    # AutoDetectExternal
    if determined and not self_flag and not external_flag and cw9_dir_exists:
        assert mode == "external", (
            f"AutoDetectExternal violated{ctx}: no flags, cw9_dir_exists=True, "
            f"determined=True but mode={mode!r}"
        )

    # AutoDetectSelf
    if determined and not self_flag and not external_flag and not cw9_dir_exists:
        assert mode == "self", (
            f"AutoDetectSelf violated{ctx}: no flags, cw9_dir_exists=False, "
            f"determined=True but mode={mode!r}"
        )
