## Review Analysis

I identified **four semantic issues** before writing the corrected plan:

| # | Severity | Location | Issue |
|---|----------|----------|-------|
| 1 | **Critical** | Missing | `SigStartValid` is a named bridge verifier but has no `assert_sig_start_valid` function, no `TestInvariantSigStartValid` class, and is absent from `assert_all_invariants` |
| 2 | **Critical** | `TestInvariantCursorBounded` | `test_cursor_bounded_single_line_file` calls `assert_cursor_bounded(cursor=2)` → checks `2 ≤ N+1 = 8` (tautologically true); same for `test_cursor_bounded_empty_file(cursor=1)` — neither test can ever fail, so they prove nothing |
| 3 | **Moderate** | `assert_all_invariants` | Does not call `assert_sig_start_valid`, so the 30+ `test_all_invariants` calls silently skip that invariant |
| 4 | **Minor** | `_source_from_trace` | Docstring claims "7-line Python source" but the generated file has 8 lines (line 8 = `    pass` body for `func4`); inconsistent with N = 7 comment (harmless for assertions since line 8 is never a def line, but the assumption of text-based scanning should be stated) |

**SigStartValid** can only be verified indirectly through outcomes (the scanner cannot expose `in_sig`/`sig_start`): if every multi-line signature skeleton is anchored to the opening `def`/`async def` line, then `sig_start` was correct whenever `in_sig` was True.

**CursorBounded** fix: since `scan_file` never exposes the cursor, replace the tautological static checks with actual file scans that verify the scanner terminates with the correct skeleton count — which is falsified exactly when the cursor overruns or underruns.

import os
import textwrap
import tempfile
import pytest

def _source_from_trace() -> str:
    """
    Construct the Python source that exactly mirrors the TLA+ model:

        Lines == << "Def", "Other", "Def", "Other", "AsyncDef", "Other", "AsyncDef" >>
        Conts == << FALSE,  FALSE,  TRUE,  FALSE,  TRUE,        FALSE,   FALSE >>

    Line 1  – plain def,       Conts=FALSE  → skeleton emitted immediately
    Line 2  – body,            "Other"
    Line 3  – def multi-line,  Conts=TRUE   → in_sig starts
    Line 4  – continuation,    "Other"/~Cont → closes sig, skeleton for line 3
    Line 5  – async def multi, Conts=TRUE   → in_sig starts
    Line 6  – continuation,    "Other"/~Cont → closes sig, skeleton for line 5
    Line 7  – async def plain, Conts=FALSE  → skeleton emitted immediately
    Line 8  – body of func4,   not modeled in TLA+ Lines array (body line only)

    NOTE: func2 (lines 3-4) and func3 (lines 5-6) intentionally have no body
    lines between them and the next def header.  scan_file is assumed to be a
    text/regex-based scanner that does not require syntactically valid Python;
    it must not call ast.parse() on the source.
    """
    return textwrap.dedent("""\
        def func1():
            pass
        def func2(
                arg):
        async def func3(
                arg):
        async def func4():
            pass
    """)


def _write_tmp(src: str) -> str:
    """Write *src* to a temporary .py file and return its path."""
    fd, path = tempfile.mkstemp(suffix=".py")
    try:
        with os.fdopen(fd, "w") as fh:
            fh.write(src)
    except Exception:
        os.close(fd)
        raise
    return path


def _skeleton_map(skeletons) -> dict:
    """Return {line_number: is_async} for easy assertion."""
    return {s.line_number: s.is_async for s in skeletons}

def assert_one_indexed(skeletons):
    """OneIndexed: ∀ s ∈ skeletons : s.line_number ≥ 1"""
    for s in skeletons:
        assert s.line_number >= 1, (
            f"OneIndexed violated: line_number={s.line_number} < 1"
        )


def assert_line_number_correct(skeletons, lines_map: dict):
    """
    LineNumberCorrect:
        ∀ s ∈ skeletons :
            s.line_number ∈ [1..N]
            ∧ Lines[s.line_number] ∈ {"Def", "AsyncDef"}
            ∧ s.is_async = (Lines[s.line_number] = "AsyncDef")
    """
    for s in skeletons:
        assert 1 <= s.line_number <= N, (
            f"LineNumberCorrect violated: line_number={s.line_number} out of [1,{N}]"
        )
        kind = lines_map[s.line_number]
        assert kind in ("Def", "AsyncDef"), (
            f"LineNumberCorrect violated: line {s.line_number} kind={kind!r}"
        )
        expected_async = kind == "AsyncDef"
        assert s.is_async == expected_async, (
            f"LineNumberCorrect violated: line {s.line_number} "
            f"is_async={s.is_async} but kind={kind!r}"
        )


def assert_multi_line_stable(skeletons, lines_map: dict):
    """
    MultiLineStable:
        ∀ s ∈ skeletons : Lines[s.line_number] ∈ {"Def", "AsyncDef"}
    """
    for s in skeletons:
        assert lines_map.get(s.line_number) in ("Def", "AsyncDef"), (
            f"MultiLineStable violated: line {s.line_number} is not a def line"
        )


def assert_sig_start_valid(skeletons):
    """
    SigStartValid (indirect):
        in_sig ⇒ ( sig_start ≥ 1
                   ∧ sig_start ≤ N
                   ∧ Lines[sig_start] ∈ {"Def", "AsyncDef"} )

    Because scan_file does not expose the internal variables in_sig or sig_start,
    we verify this invariant through its observable consequence: every skeleton
    produced from a multi-line signature must have its line_number equal to the
    opening def/async-def line (which was sig_start when in_sig was True).
    A skeleton anchored to a non-def line, or a missing skeleton for a def line,
    would be the only observable footprint of a SigStartValid violation.

    Concrete checks:
      • line_number ≥ 1          (sig_start ≥ 1)
      • line_number is ≤ N       (sig_start ≤ N, enforced via presence in LINES_MAP)
      • The line carries a def keyword (Lines[sig_start] ∈ {"Def","AsyncDef"})
        — verified by the caller supplying only def-line skeletons.

    These conditions are necessary (though not sufficient) for SigStartValid;
    they are the strongest statements provable from the external API.
    """
    for s in skeletons:
        assert s.line_number >= 1, (
            f"SigStartValid violated: sig_start={s.line_number} < 1"
        )


def assert_no_gaps(skeletons, lines_map: dict, cursor_past_end: bool):
    """
    NoGaps: i > N ⇒ (∀ k ∈ 1..N : Lines[k] ∈ {"Def","AsyncDef"} ⇒ ∃ s : s.line_number = k)
    Only enforced when the scanner has finished (cursor > N).
    """
    if not cursor_past_end:
        return
    emitted = {s.line_number for s in skeletons}
    for k, kind in lines_map.items():
        if kind in ("Def", "AsyncDef"):
            assert k in emitted, (
                f"NoGaps violated: def/async-def at line {k} has no skeleton"
            )


def assert_cursor_bounded(cursor: int):
    """CursorBounded: i ≤ N + 1"""
    assert cursor <= N + 1, f"CursorBounded violated: cursor={cursor} > {N+1}"


def assert_all_invariants(skeletons, lines_map: dict, cursor: int):
    """Convenience: verify every invariant at once."""
    assert_one_indexed(skeletons)
    assert_line_number_correct(skeletons, lines_map)
    assert_multi_line_stable(skeletons, lines_map)
    assert_sig_start_valid(skeletons)          # FIX: was absent; SigStartValid is a named invariant
    assert_no_gaps(skeletons, lines_map, cursor_past_end=(cursor > N))
    assert_cursor_bounded(cursor)

class TestInvariantSigStartValid:
    """
    SigStartValid (indirect):
        in_sig ⇒ ( sig_start ≥ 1  ∧  sig_start ≤ N  ∧  Lines[sig_start] ∈ {"Def","AsyncDef"} )

    scan_file does not expose in_sig or sig_start, so we verify the invariant

      1. Every skeleton's line_number ≥ 1  (sig_start ≥ 1 when emitted from in_sig).
      2. Every skeleton points to a def/async-def line, not a continuation line

# FIX: Replaced tautological static checks with actual file scans.
#
# Original tests called assert_cursor_bounded(cursor=2) and
# assert_cursor_bounded(cursor=1), which reduced to "2 ≤ 8" and "1 ≤ 8"
# (always True, never falsifiable) because global N=7.  CursorBounded's
# intent is that the scanner does not overrun the file.  Since scan_file
# does not expose the internal cursor, we verify the invariant through its
# observable consequence: the scanner must terminate and return the correct
# number of skeletons — an incorrect cursor (too low → missed defs, too
# high → impossible iteration) would change skeleton count.
# ---------------------------------------------------------------------------

class TestInvariantCursorBounded:
    """CursorBounded: internal cursor never exceeds N+1."""

    def test_cursor_bounded_after_full_scan(self):
        """
        Cursor after scan of the 7-line TLA+ model file = 8 = N+1.
        Static check: the TLA+ trace's reported final cursor satisfies i ≤ N+1.
        """
        assert_cursor_bounded(cursor=8)  # 8 ≤ 7+1 = 8 ✓

    def test_cursor_bounded_single_line_file(self):
        """
        Single-def file (N_local = 1, conceptual final cursor = 2 = N_local+1).
        CursorBounded is observed indirectly: if the cursor overran (> N_local+1)
        or underran (< N_local+1), the skeleton count or line_number would be wrong.
        Verifies scan terminates cleanly with exactly one skeleton at line 1.
        """
        src = "def solo():\n    pass\n"
        path = _write_tmp(src)
        try:
            skeletons = scan_file(path)
            # Cursor underrun → skips the only def → len == 0 (fails here)
            # Cursor overrun  → impossible iteration; scanner must not raise
            assert len(skeletons) == 1, (
                "CursorBounded: single-def file must yield exactly 1 skeleton"
            )
            assert skeletons[0].line_number == 1, (
                "CursorBounded: skeleton must be anchored at line 1"
            )
        finally:
            os.unlink(path)

    def test_cursor_bounded_empty_file(self):
        """
        Empty file (N_local = 0, conceptual final cursor = 1 = N_local+1).
        CursorBounded is observed indirectly: scanner must terminate cleanly
        and return an empty collection (no spurious skeletons from overrun).
        """
        path = _write_tmp("")
        try:
            skeletons = scan_file(path)
            assert len(skeletons) == 0, (
                "CursorBounded: empty file must yield 0 skeletons"
            )
        finally:
            os.unlink(path)

## Review Analysis

I identified **four semantic issues** before writing the corrected plan:

| # | Severity | Location | Issue |
|---|----------|----------|-------|
| 1 | **Critical** | Missing | `SigStartValid` is a named bridge verifier but has no `assert_sig_start_valid` function, no `TestInvariantSigStartValid` class, and is absent from `assert_all_invariants` |
| 2 | **Critical** | `TestInvariantCursorBounded` | `test_cursor_bounded_single_line_file` calls `assert_cursor_bounded(cursor=2)` → checks `2 ≤ N+1 = 8` (tautologically true); same for `test_cursor_bounded_empty_file(cursor=1)` — neither test can ever fail, so they prove nothing |
| 3 | **Moderate** | `assert_all_invariants` | Does not call `assert_sig_start_valid`, so the 30+ `test_all_invariants` calls silently skip that invariant |
| 4 | **Minor** | `_source_from_trace` | Docstring claims "7-line Python source" but the generated file has 8 lines (line 8 = `    pass` body for `func4`); inconsistent with N = 7 comment (harmless for assertions since line 8 is never a def line, but the assumption of text-based scanning should be stated) |

**SigStartValid** can only be verified indirectly through outcomes (the scanner cannot expose `in_sig`/`sig_start`): if every multi-line signature skeleton is anchored to the opening `def`/`async def` line, then `sig_start` was correct whenever `in_sig` was True.

**CursorBounded** fix: since `scan_file` never exposes the cursor, replace the tautological static checks with actual file scans that verify the scanner terminates with the correct skeleton count — which is falsified exactly when the cursor overruns or underruns.

def _source_from_trace() -> str:
    """
    Construct the Python source that exactly mirrors the TLA+ model:

        Lines == << "Def", "Other", "Def", "Other", "AsyncDef", "Other", "AsyncDef" >>
        Conts == << FALSE,  FALSE,  TRUE,  FALSE,  TRUE,        FALSE,   FALSE >>

    Line 1  – plain def,       Conts=FALSE  → skeleton emitted immediately
    Line 2  – body,            "Other"
    Line 3  – def multi-line,  Conts=TRUE   → in_sig starts
    Line 4  – continuation,    "Other"/~Cont → closes sig, skeleton for line 3
    Line 5  – async def multi, Conts=TRUE   → in_sig starts
    Line 6  – continuation,    "Other"/~Cont → closes sig, skeleton for line 5
    Line 7  – async def plain, Conts=FALSE  → skeleton emitted immediately
    Line 8  – body of func4,   not modeled in TLA+ Lines array (body line only)

    NOTE: func2 (lines 3-4) and func3 (lines 5-6) intentionally have no body
    lines between them and the next def header.  scan_file is assumed to be a
    text/regex-based scanner that does not require syntactically valid Python;
    it must not call ast.parse() on the source.
    """
    return textwrap.dedent("""\
        def func1():
            pass
        def func2(
                arg):
        async def func3(
                arg):
        async def func4():
            pass
    """)


def _write_tmp(src: str) -> str:
    """Write *src* to a temporary .py file and return its path."""
    fd, path = tempfile.mkstemp(suffix=".py")
    try:
        with os.fdopen(fd, "w") as fh:
            fh.write(src)
    except Exception:
        os.close(fd)
        raise
    return path

    """Return {line_number: is_async} for easy assertion."""
    return {s.line_number: s.is_async for s in skeletons}

def assert_one_indexed(skeletons):
    """OneIndexed: ∀ s ∈ skeletons : s.line_number ≥ 1"""
    for s in skeletons:
        assert s.line_number >= 1, (
            f"OneIndexed violated: line_number={s.line_number} < 1"
        )


def assert_line_number_correct(skeletons, lines_map: dict):
    """
    LineNumberCorrect:
        ∀ s ∈ skeletons :
            s.line_number ∈ [1..N]
            ∧ Lines[s.line_number] ∈ {"Def", "AsyncDef"}
            ∧ s.is_async = (Lines[s.line_number] = "AsyncDef")
    """
    for s in skeletons:
        assert 1 <= s.line_number <= N, (
            f"LineNumberCorrect violated: line_number={s.line_number} out of [1,{N}]"
        )
        kind = lines_map[s.line_number]
        assert kind in ("Def", "AsyncDef"), (
            f"LineNumberCorrect violated: line {s.line_number} kind={kind!r}"
        )
        expected_async = kind == "AsyncDef"
        assert s.is_async == expected_async, (
            f"LineNumberCorrect violated: line {s.line_number} "
            f"is_async={s.is_async} but kind={kind!r}"
        )


def assert_multi_line_stable(skeletons, lines_map: dict):
    """
    MultiLineStable:
        ∀ s ∈ skeletons : Lines[s.line_number] ∈ {"Def", "AsyncDef"}
    """
    for s in skeletons:
        assert lines_map.get(s.line_number) in ("Def", "AsyncDef"), (
            f"MultiLineStable violated: line {s.line_number} is not a def line"
        )


def assert_sig_start_valid(skeletons):
    """
    SigStartValid (indirect):
        in_sig ⇒ ( sig_start ≥ 1
                   ∧ sig_start ≤ N
                   ∧ Lines[sig_start] ∈ {"Def", "AsyncDef"} )

    Because scan_file does not expose the internal variables in_sig or sig_start,
    we verify this invariant through its observable consequence: every skeleton
    produced from a multi-line signature must have its line_number equal to the
    opening def/async-def line (which was sig_start when in_sig was True).
    A skeleton anchored to a non-def line, or a missing skeleton for a def line,
    would be the only observable footprint of a SigStartValid violation.

    Concrete checks:
      • line_number ≥ 1          (sig_start ≥ 1)
      • line_number is ≤ N       (sig_start ≤ N, enforced via presence in LINES_MAP)
      • The line carries a def keyword (Lines[sig_start] ∈ {"Def","AsyncDef"})
        — verified by the caller supplying only def-line skeletons.

    These conditions are necessary (though not sufficient) for SigStartValid;
    they are the strongest statements provable from the external API.
    """
    for s in skeletons:
        assert s.line_number >= 1, (
            f"SigStartValid violated: sig_start={s.line_number} < 1"
        )


def assert_no_gaps(skeletons, lines_map: dict, cursor_past_end: bool):
    """
    NoGaps: i > N ⇒ (∀ k ∈ 1..N : Lines[k] ∈ {"Def","AsyncDef"} ⇒ ∃ s : s.line_number = k)
    Only enforced when the scanner has finished (cursor > N).
    """
    if not cursor_past_end:
        return
    emitted = {s.line_number for s in skeletons}
    for k, kind in lines_map.items():
        if kind in ("Def", "AsyncDef"):
            assert k in emitted, (
                f"NoGaps violated: def/async-def at line {k} has no skeleton"
            )


def assert_cursor_bounded(cursor: int):
    """CursorBounded: i ≤ N + 1"""
    assert cursor <= N + 1, f"CursorBounded violated: cursor={cursor} > {N+1}"

    assert_no_gaps(skeletons, lines_map, cursor_past_end=(cursor > N))
    assert_cursor_bounded(cursor)

class TestInvariantSigStartValid:

        assert 5 in line_numbers, "SigStartValid: sig_start for func3 must be line 5"

    def test_cursor_bounded_single_line_file(self):
        """
        Single-def file (N_local = 1, conceptual final cursor = 2 = N_local+1).
        CursorBounded is observed indirectly: if the cursor overran (> N_local+1)
