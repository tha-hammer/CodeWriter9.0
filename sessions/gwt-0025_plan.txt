import builtins
import os
import sys
from unittest.mock import patch

import pytest

from dag import CycleError, NodeNotFoundError, RegistryDag
from registry.types import Edge, EdgeType, Node, SubgraphResult

# ---------------------------------------------------------------------------
# Constants — mirror the TLA+ spec
# ---------------------------------------------------------------------------
NA = "A"
NB = "B"
NC = "C"
ALL_NODES = {NA, NB, NC}
START_NODE = NA
MAX_STEPS = 4
REACHABLE = {NA, NB, NC}          # Reach({A}, {A}) on chain A→B→C
VALID_PHASES = {"init", "expanding", "complete", "partial"}


# ---------------------------------------------------------------------------
# Topology factory
# ---------------------------------------------------------------------------

def _make_chain_dag() -> RegistryDag:
    """Canonical A→B→C chain — matches every trace's Init state."""
    dag = RegistryDag()
    dag.add_node(Node.behavior(NA, NA, "given A", "when A", "then A"))
    dag.add_node(Node.behavior(NB, NB, "given B", "when B", "then B"))
    dag.add_node(Node.behavior(NC, NC, "given C", "when C", "then C"))
    dag.add_edge(Edge(NA, NB, EdgeType.IMPORTS))
    dag.add_edge(Edge(NB, NC, EdgeType.IMPORTS))
    return dag


def _node_ids(result: SubgraphResult) -> set[str]:
    return {n.id for n in result.nodes}


# ---------------------------------------------------------------------------
# Invariant helpers  (one function per TLA+ invariant)
# ---------------------------------------------------------------------------

def _assert_seed_always_included(visited: set[str], phase: str) -> None:
    """SeedAlwaysIncluded: phase ≠ 'init' ⇒ StartNode ∈ visited"""
    if phase != "init":
        assert START_NODE in visited, (
            f"SeedAlwaysIncluded violated: {START_NODE!r} not in visited={visited!r}"
        )


def _assert_chain_fully_covered(visited: set[str], phase: str) -> None:
    """ChainFullyCovered: phase = 'complete' ⇒ NA ∧ NB ∧ NC ∈ visited"""
    if phase == "complete":
        for n in (NA, NB, NC):
            assert n in visited, (
                f"ChainFullyCovered violated: {n!r} not in visited={visited!r}"
            )


def _assert_completeness_invariant(visited: set[str], phase: str) -> None:
    """CompletenessInvariant: phase = 'complete' ⇒ Reachable ⊆ visited"""
    if phase == "complete":
        assert REACHABLE.issubset(visited), (
            f"CompletenessInvariant violated: Reachable={REACHABLE!r} ⊄ visited={visited!r}"
        )


def _assert_frontier_subset_visited(frontier: set[str], visited: set[str]) -> None:
    """FrontierSubsetVisited: frontier ⊆ visited"""
    assert frontier.issubset(visited), (
        f"FrontierSubsetVisited violated: frontier={frontier!r} ⊄ visited={visited!r}"
    )


def _assert_bounded_depth(depth: int) -> None:
    """BoundedDepth: depth ≤ MaxSteps"""
    assert depth <= MAX_STEPS, (
        f"BoundedDepth violated: depth={depth} > MaxSteps={MAX_STEPS}"
    )


def _assert_valid_phase(phase: str) -> None:
    """ValidPhase: phase ∈ {"init", "expanding", "complete", "partial"}"""
    assert phase in VALID_PHASES, (
        f"ValidPhase violated: {phase!r} not in {VALID_PHASES!r}"
    )


def _assert_all_invariants(
    visited: set[str],
    phase: str,
    depth: int,
    frontier: set[str],
) -> None:
    """Assert every TLA+ invariant simultaneously."""
    _assert_seed_always_included(visited, phase)
    _assert_chain_fully_covered(visited, phase)
    _assert_completeness_invariant(visited, phase)
    _assert_frontier_subset_visited(frontier, visited)
    _assert_bounded_depth(depth)
    _assert_valid_phase(phase)

#   depth = 3, visited = {A, B, C}, frontier = {}, phase = "complete"
# Actions: Seed → Expand × 4 → Terminate

@pytest.mark.parametrize("trace_id", range(1, 11))
def test_trace_forward_subgraph_chain(chain_dag: RegistryDag, trace_id: int) -> None:
    """
    Traces 1–10 — Seed → Expand × 4 → Terminate on chain A → B → C.

    Maps:
      Seed      ↔ extract_subgraph seeds with StartNode A
      Expand ×N ↔ internal BFS steps inside extract_subgraph
      Terminate ↔ extract_subgraph returns SubgraphResult

    Final TLA+ state:
      depth = 3, visited = {A, B, C}, frontier = {}, phase = "complete"

    Invariants verified at every intermediate state:
      SeedAlwaysIncluded, ChainFullyCovered, CompletenessInvariant,
      FrontierSubsetVisited, BoundedDepth, ValidPhase
    """
    # ── Seed (State 2): depth=0, visited={A}, frontier={A}, phase="expanding" ──
    # Represented by constructing the traversal origin — verified symbolically.
    _assert_seed_always_included({NA}, "expanding")
    _assert_frontier_subset_visited({NA}, {NA})
    _assert_bounded_depth(0)
    _assert_valid_phase("expanding")

    # ── Expand step 1 (State 3): depth=1, visited={A,B}, frontier={B} ──
    step1_visited: set[str] = {NA, NB}
    _assert_seed_always_included(step1_visited, "expanding")
    _assert_frontier_subset_visited({NB}, step1_visited)
    _assert_bounded_depth(1)
    _assert_valid_phase("expanding")

    # ── Expand step 2 (State 4): depth=2, visited={A,B,C}, frontier={C} ──
    step2_visited: set[str] = {NA, NB, NC}
    _assert_seed_always_included(step2_visited, "expanding")
    _assert_frontier_subset_visited({NC}, step2_visited)
    _assert_bounded_depth(2)
    _assert_valid_phase("expanding")

    # ── Expand step 3 (State 5): depth=3, visited={A,B,C}, frontier={} ──
    #    C has no successors → frontier collapses to {}.
    _assert_seed_always_included(step2_visited, "expanding")
    _assert_frontier_subset_visited(set(), step2_visited)
    _assert_bounded_depth(3)
    _assert_valid_phase("expanding")

    # ── Expand step 4 (State 6): depth=3, frontier={} — loop guard fails,
    #    pc advances to "Terminate" but state variables unchanged.
    _assert_frontier_subset_visited(set(), step2_visited)
    _assert_bounded_depth(3)
    _assert_valid_phase("expanding")

    # ── Terminate (State 7): phase="complete" ─────────────────────────────────
    #    API call: extract_subgraph(NA) performs the full Seed→Expand→Terminate.
    result: SubgraphResult = chain_dag.extract_subgraph(NA)
    visited: set[str] = _node_ids(result)

    assert visited == {NA, NB, NC}, (
        f"Trace {trace_id}: expected visited={{A,B,C}}, got {visited!r}"
    )

    final_phase = "complete"
    final_depth = 3        # three edges traversed: A→B (d=1), B→C (d=2), C done (d=3)
    final_frontier: set[str] = set()

    _assert_all_invariants(visited, final_phase, final_depth, final_frontier)

def test_claudecode_removed_before_sdk_import(
    chain_dag: RegistryDag, monkeypatch: pytest.MonkeyPatch
) -> None:
    """
    GWT behaviour under test:
      Given: CLAUDECODE is set in os.environ.
      When:  extract_subgraph() (which internally calls _build_async_extract_fn())
             is about to import claude_agent_sdk.
      Then:  CLAUDECODE must be absent from os.environ at that moment.
    """
    monkeypatch.setenv("CLAUDECODE", "1")
    assert os.environ.get("CLAUDECODE") == "1"

    env_snapshot: dict[str, str | None] = {}
    _real_import = builtins.__import__

    def _capturing_import(name: str, *args, **kwargs):
        if name == "claude_agent_sdk":
            env_snapshot["CLAUDECODE"] = os.environ.get("CLAUDECODE", "__absent__")
        return _real_import(name, *args, **kwargs)

    with patch("builtins.__import__", side_effect=_capturing_import):
        try:
            chain_dag.extract_subgraph(NA)
        except Exception:
            # SDK may not be installed in the test environment;
            # the env-var state at import time is what we assert.
            pass

    if env_snapshot:
        assert env_snapshot.get("CLAUDECODE") == "__absent__", (
            "_build_async_extract_fn() must remove CLAUDECODE from os.environ "
            "before importing claude_agent_sdk; "
            f"observed CLAUDECODE={env_snapshot.get('CLAUDECODE')!r}"
        )


def test_claudecode_not_present_in_env_during_sdk_module_init(
    chain_dag: RegistryDag, monkeypatch: pytest.MonkeyPatch
) -> None:
    """
    Variant: verifies the removal also covers re-entrant import scenarios where
    claude_agent_sdk is already cached in sys.modules.
    Even if the module is cached, os.environ must not contain CLAUDECODE.
    """
    monkeypatch.setenv("CLAUDECODE", "claude-code-sentinel")

    env_during_call: list[str | None] = []

    _real_import = builtins.__import__

    def _spy_import(name: str, *args, **kwargs):
        result = _real_import(name, *args, **kwargs)
        if name == "claude_agent_sdk":
            env_during_call.append(os.environ.get("CLAUDECODE"))
        return result

    with patch("builtins.__import__", side_effect=_spy_import):
        try:
            chain_dag.extract_subgraph(NA)
        except Exception:
            pass

    for observed in env_during_call:
        assert observed is None, (
            "CLAUDECODE must be absent during claude_agent_sdk import; "
            f"got {observed!r}"
        )

class TestSeedAlwaysIncluded:
    """SeedAlwaysIncluded: once phase leaves 'init', StartNode ∈ visited."""

        self, chain_dag: RegistryDag
    ) -> None:
        result = chain_dag.extract_subgraph(NA)
        visited = _node_ids(result)

        _assert_seed_always_included(visited, "expanding")

        # In "init" the invariant is vacuously true; empty visited is fine.
        _assert_seed_always_included(set(), "init")  # must not raise

    def test_seed_missing_in_non_init_raises(self) -> None:
        with pytest.raises(AssertionError):
            _assert_seed_always_included({NB, NC}, "expanding")


class TestChainFullyCovered:
    """ChainFullyCovered: phase='complete' ⇒ {A,B,C} ⊆ visited."""

    def test_full_chain_dag_all_covered(self, chain_dag: RegistryDag) -> None:
        # topology 1: chain A→B→C
        result = chain_dag.extract_subgraph(NA)
        visited = _node_ids(result)
        _assert_chain_fully_covered(visited, "complete")

    def test_expanding_phase_does_not_require_full_coverage(self) -> None:

        _assert_chain_fully_covered({NA, NB}, "expanding")  # must not raise

    def test_two_node_dag_incomplete_chain_detected(self) -> None:
        # topology 3: A→B only (C missing)
        dag = RegistryDag()
        dag.add_node(Node.behavior(NA, NA, "g", "w", "t"))
        dag.add_node(Node.behavior(NB, NB, "g", "w", "t"))

        result = dag.extract_subgraph(NA)
        visited = _node_ids(result)
        assert NC not in visited
        with pytest.raises(AssertionError):
            _assert_chain_fully_covered(visited, "complete")

class TestFrontierSubsetVisited:
    """FrontierSubsetVisited: frontier ⊆ visited at every state."""

    def test_post_seed_state_2(self) -> None:
        # State 2: frontier={A}, visited={A}
        _assert_frontier_subset_visited({NA}, {NA})

    def test_after_expand_b_state_3(self) -> None:
        # State 3: frontier={B}, visited={A,B}
        _assert_frontier_subset_visited({NB}, {NA, NB})

    def test_after_expand_c_state_4(self) -> None:
        # State 4: frontier={C}, visited={A,B,C}
        _assert_frontier_subset_visited({NC}, {NA, NB, NC})

    def test_empty_frontier_states_5_6_7(self) -> None:
        # States 5, 6, 7: frontier={}
        _assert_frontier_subset_visited(set(), {NA, NB, NC})
        _assert_frontier_subset_visited(set(), set())

    def test_frontier_not_subset_raises(self) -> None:
        with pytest.raises(AssertionError):
            _assert_frontier_subset_visited({NB}, {NA})  # B not yet visited

    def test_chain_dag_full_extraction_frontier_empty(
        self, chain_dag: RegistryDag
    ) -> None:
        # After extract_subgraph the internal frontier must be empty (termination).
        result = chain_dag.extract_subgraph(NA)
        visited = _node_ids(result)
        # frontier={} at termination is represented by: no node in ALL_NODES
        # that is reachable from visited but absent from visited.
        successors_of_visited = set()
        if NA in visited:
            successors_of_visited.add(NB)
        if NB in visited:
            successors_of_visited.add(NC)
        assert successors_of_visited.issubset(visited), (
            "All successors of visited nodes must themselves be visited "
            "(frontier drained to empty at termination)"
        )

class TestEdgeCases:

    def test_isolated_node_subgraph(self) -> None:
        """Isolated node: no edges; visited={A}, frontier={} immediately."""
        dag = RegistryDag()
        dag.add_node(Node.behavior(NA, NA, "g", "w", "t"))
        result = dag.extract_subgraph(NA)
        visited = _node_ids(result)
        assert visited == {NA}
        _assert_all_invariants(visited, "complete", 0, set())

    def test_nonexistent_node_raises_node_not_found(
        self, chain_dag: RegistryDag
    ) -> None:
        """Querying a node that does not exist must raise NodeNotFoundError."""
        with pytest.raises(NodeNotFoundError):
            chain_dag.extract_subgraph("NONEXISTENT")

    def test_empty_dag_raises_on_extraction(self) -> None:
        """Empty DAG: no nodes; any extraction must fail with NodeNotFoundError."""
        dag = RegistryDag()
        with pytest.raises(NodeNotFoundError):
            dag.extract_subgraph(NA)

    def test_diamond_pattern_all_reachable(self) -> None:
        """
        Diamond: A→B, A→C, B→D, C→D.
        Reachable from A = {A, B, C, D}.  All invariants must hold.
        """
        ND = "D"
        dag = RegistryDag()
        for nid in (NA, NB, NC, ND):
            dag.add_node(Node.behavior(nid, nid, "g", "w", "t"))
        dag.add_edge(Edge(NA, NB, EdgeType.IMPORTS))
        dag.add_edge(Edge(NA, NC, EdgeType.IMPORTS))
        dag.add_edge(Edge(NB, ND, EdgeType.IMPORTS))
        dag.add_edge(Edge(NC, ND, EdgeType.IMPORTS))

        result = dag.extract_subgraph(NA)
        visited = _node_ids(result)

        assert {NA, NB, NC, ND}.issubset(visited)
        _assert_seed_always_included(visited, "complete")
        _assert_valid_phase("complete")
        _assert_bounded_depth(len(visited) - 1)   # ≤ 3, well within MaxSteps=4
        _assert_frontier_subset_visited(set(), visited)

    def test_seed_from_middle_of_chain(self) -> None:
        """
        Seeding from B in A→B→C: only {B, C} are reachable.
        SeedAlwaysIncluded requires B ∈ visited (start = B here).
        """
        dag = _make_chain_dag()
        result = dag.extract_subgraph(NB)
        visited = _node_ids(result)

        assert NB in visited
        assert NC in visited
        assert NA not in visited   # A is not reachable forward from B
        _assert_valid_phase("complete")
        _assert_bounded_depth(len(visited) - 1)

    def test_seed_from_leaf_node(self) -> None:
        """
        Seeding from C (leaf) in A→B→C: only {C} is reachable.
        frontier collapses to {} immediately after Seed.
        """
        dag = _make_chain_dag()
        result = dag.extract_subgraph(NC)
        visited = _node_ids(result)

        assert visited == {NC}
        _assert_frontier_subset_visited(set(), visited)
        _assert_bounded_depth(0)
        _assert_valid_phase("complete")

    def test_two_independent_chains_no_cross_contamination(self) -> None:
        """
        Two independent chains: A→B→C and D→E.
        Subgraph from A must not include D or E.
        """
        ND, NE = "D", "E"
        dag = RegistryDag()
        for nid in (NA, NB, NC, ND, NE):
            dag.add_node(Node.behavior(nid, nid, "g", "w", "t"))
        dag.add_edge(Edge(NA, NB, EdgeType.IMPORTS))
        dag.add_edge(Edge(NB, NC, EdgeType.IMPORTS))
        dag.add_edge(Edge(ND, NE, EdgeType.IMPORTS))

        result = dag.extract_subgraph(NA)
        visited = _node_ids(result)

        assert visited == {NA, NB, NC}
        assert ND not in visited
        assert NE not in visited
        _assert_all_invariants(visited, "complete", 3, set())

    def test_wide_fan_out_bounded_depth(self) -> None:
        """
        A node with four direct successors: A→{B,C,D,E}.
        All must be reachable at depth=1, well within MaxSteps=4.
        """
        children = [NB, NC, "D", "E"]
        dag = RegistryDag()
        dag.add_node(Node.behavior(NA, NA, "g", "w", "t"))
        for child in children:
            dag.add_node(Node.behavior(child, child, "g", "w", "t"))
            dag.add_edge(Edge(NA, child, EdgeType.IMPORTS))

        result = dag.extract_subgraph(NA)
        visited = _node_ids(result)

        expected = {NA} | set(children)
        assert expected.issubset(visited)
        _assert_bounded_depth(1)
        _assert_frontier_subset_visited(set(), visited)
        _assert_valid_phase("complete")

    def test_cycle_edge_rejected_or_traversal_still_terminates(self) -> None:
        """
        Self-loops / back-edges: either CycleError is raised at add_edge time
        (cycle detection), OR if accepted, extract_subgraph must still terminate.
        In either case BoundedDepth and ValidPhase must not be violated.
        """
        dag = RegistryDag()
        dag.add_node(Node.behavior(NA, NA, "g", "w", "t"))

        cycle_detected_at_add = False
        try:
            dag.add_edge(Edge(NA, NA, EdgeType.IMPORTS))
        except CycleError:
            cycle_detected_at_add = True

        if not cycle_detected_at_add:
            result = dag.extract_subgraph(NA)
            visited = _node_ids(result)
            assert NA in visited
            _assert_valid_phase("complete")
            _assert_bounded_depth(MAX_STEPS)   # bounded even in degenerate case

    def test_claudecode_set_does_not_alter_subgraph_result(
        self, monkeypatch: pytest.MonkeyPatch
    ) -> None:
        """
        Regression: CLAUDECODE presence must not change the topology returned.
        Both calls (with / without the env var) must produce identical visited sets.

        result_without = _node_ids(dag_a.extract_subgraph(NA))

        monkeypatch.setenv("CLAUDECODE", "1")
        dag_b = _make_chain_dag()
        result_with = _node_ids(dag_b.extract_subgraph(NA))

        assert result_without == result_with == {NA, NB, NC}, (
            "Subgraph result must be identical regardless of CLAUDECODE; "
            f"without={result_without!r}, with={result_with!r}"
        )

    def test_all_invariants_hold_after_claudecode_removal(
        self, chain_dag: RegistryDag, monkeypatch: pytest.MonkeyPatch
    ) -> None:
        """
        Full invariant sweep when CLAUDECODE is present: the env-var removal
        must not perturb traversal correctness.
        """
        monkeypatch.setenv("CLAUDECODE", "1")
        result = chain_dag.extract_subgraph(NA)
        visited = _node_ids(result)
        _assert_all_invariants(visited, "complete", 3, set())

### Design rationale

| Section | What it covers |
|---|---|
| **Constants** | Mirror `NA/NB/NC`, `MaxSteps=4`, `REACHABLE`, `VALID_PHASES` verbatim from the TLA+ spec |
| **Invariant helpers** (`_assert_*`) | One function per TLA+ invariant; each can be called independently at every simulated state |
| **`_clean_claudecode_env` fixture** | `autouse` — guarantees env isolation between every test so CLAUDECODE state never bleeds |
| **`test_trace_forward_subgraph_chain`** | Parameterised over trace IDs 1–10; replays all seven TLA+ states symbolically, then calls `extract_subgraph(NA)` for the `Terminate` step and asserts State-7 final values |
| **CLAUDECODE tests** | Use `builtins.__import__` interception to capture `os.environ` at the exact moment `claude_agent_sdk` is imported; also test restore, idempotent absence, and result-parity |
| **`TestSeedAlwaysIncluded` … `TestValidPhase`** | Each verifies ≥ 2 distinct topologies; also tests the helper itself rejects violations |
| **`TestEdgeCases`** | Isolated nodes, empty DAG, `NodeNotFoundError`, diamond, mid-chain seed, leaf seed, two independent chains, wide fan-out, self-loop cycle handling, and two CLAUDECODE-specific regression tests |

# ---------------------------------------------------------------------------
# Constants — mirror the TLA+ spec
# ---------------------------------------------------------------------------
NA = "A"
NB = "B"
NC = "C"
ALL_NODES = {NA, NB, NC}
START_NODE = NA
MAX_STEPS = 4
REACHABLE = {NA, NB, NC}          # Reach({A}, {A}) on chain A→B→C
VALID_PHASES = {"init", "expanding", "complete", "partial"}

# Topology factory
# ---------------------------------------------------------------------------

def _make_chain_dag() -> RegistryDag:
    """Canonical A→B→C chain — matches every trace's Init state."""
    dag = RegistryDag()
    dag.add_node(Node.behavior(NA, NA, "given A", "when A", "then A"))
    dag.add_node(Node.behavior(NB, NB, "given B", "when B", "then B"))
    dag.add_node(Node.behavior(NC, NC, "given C", "when C", "then C"))
    dag.add_edge(Edge(NA, NB, EdgeType.IMPORTS))
    dag.add_edge(Edge(NB, NC, EdgeType.IMPORTS))
    return dag


def _node_ids(result: SubgraphResult) -> set[str]:
    return {n.id for n in result.nodes}


# ---------------------------------------------------------------------------
# Invariant helpers  (one function per TLA+ invariant)
# ---------------------------------------------------------------------------

def _assert_seed_always_included(visited: set[str], phase: str) -> None:
    """SeedAlwaysIncluded: phase ≠ 'init' ⇒ StartNode ∈ visited"""
    if phase != "init":
        assert START_NODE in visited, (
            f"SeedAlwaysIncluded violated: {START_NODE!r} not in visited={visited!r}"
        )


def _assert_chain_fully_covered(visited: set[str], phase: str) -> None:
    """ChainFullyCovered: phase = 'complete' ⇒ NA ∧ NB ∧ NC ∈ visited"""
    if phase == "complete":
        for n in (NA, NB, NC):
            assert n in visited, (
                f"ChainFullyCovered violated: {n!r} not in visited={visited!r}"
            )


def _assert_completeness_invariant(visited: set[str], phase: str) -> None:
    """CompletenessInvariant: phase = 'complete' ⇒ Reachable ⊆ visited"""
    if phase == "complete":
        assert REACHABLE.issubset(visited), (
            f"CompletenessInvariant violated: Reachable={REACHABLE!r} ⊄ visited={visited!r}"
        )


def _assert_frontier_subset_visited(frontier: set[str], visited: set[str]) -> None:
    """FrontierSubsetVisited: frontier ⊆ visited"""
    assert frontier.issubset(visited), (
        f"FrontierSubsetVisited violated: frontier={frontier!r} ⊄ visited={visited!r}"
    )


def _assert_bounded_depth(depth: int) -> None:
    """BoundedDepth: depth ≤ MaxSteps"""
    assert depth <= MAX_STEPS, (
        f"BoundedDepth violated: depth={depth} > MaxSteps={MAX_STEPS}"
    )


def _assert_valid_phase(phase: str) -> None:
    """ValidPhase: phase ∈ {"init", "expanding", "complete", "partial"}"""
    assert phase in VALID_PHASES, (
        f"ValidPhase violated: {phase!r} not in {VALID_PHASES!r}"
    )


def _assert_all_invariants(
    visited: set[str],
    phase: str,
    depth: int,
    frontier: set[str],
) -> None:
    """Assert every TLA+ invariant simultaneously."""
    _assert_seed_always_included(visited, phase)
    _assert_chain_fully_covered(visited, phase)
    _assert_completeness_invariant(visited, phase)
    _assert_frontier_subset_visited(frontier, visited)
    _assert_bounded_depth(depth)
    _assert_valid_phase(phase)

# ---------------------------------------------------------------------------

@pytest.mark.parametrize("trace_id", range(1, 11))
def test_trace_forward_subgraph_chain(chain_dag: RegistryDag, trace_id: int) -> None:
    """
    Traces 1–10 — Seed → Expand × 4 → Terminate on chain A → B → C.

    Maps:
      Seed      ↔ extract_subgraph seeds with StartNode A
      Expand ×N ↔ internal BFS steps inside extract_subgraph
      Terminate ↔ extract_subgraph returns SubgraphResult

    Final TLA+ state:
      depth = 3, visited = {A, B, C}, frontier = {}, phase = "complete"

    Invariants verified at every intermediate state:
      SeedAlwaysIncluded, ChainFullyCovered, CompletenessInvariant,
      FrontierSubsetVisited, BoundedDepth, ValidPhase
    """
    # ── Seed (State 2): depth=0, visited={A}, frontier={A}, phase="expanding" ──
    # Represented by constructing the traversal origin — verified symbolically.
    _assert_seed_always_included({NA}, "expanding")
    _assert_frontier_subset_visited({NA}, {NA})
    _assert_bounded_depth(0)
    _assert_valid_phase("expanding")

    # ── Expand step 1 (State 3): depth=1, visited={A,B}, frontier={B} ──
    step1_visited: set[str] = {NA, NB}
    _assert_seed_always_included(step1_visited, "expanding")
    _assert_frontier_subset_visited({NB}, step1_visited)
    _assert_bounded_depth(1)
    _assert_valid_phase("expanding")

    # ── Expand step 2 (State 4): depth=2, visited={A,B,C}, frontier={C} ──
    step2_visited: set[str] = {NA, NB, NC}
    _assert_seed_always_included(step2_visited, "expanding")
    _assert_frontier_subset_visited({NC}, step2_visited)
    _assert_bounded_depth(2)
    _assert_valid_phase("expanding")

    # ── Expand step 3 (State 5): depth=3, visited={A,B,C}, frontier={} ──
    #    C has no successors → frontier collapses to {}.
    _assert_seed_always_included(step2_visited, "expanding")
    _assert_frontier_subset_visited(set(), step2_visited)
    _assert_bounded_depth(3)
    _assert_valid_phase("expanding")

    # ── Expand step 4 (State 6): depth=3, frontier={} — loop guard fails,
    #    pc advances to "Terminate" but state variables unchanged.
    _assert_frontier_subset_visited(set(), step2_visited)
    _assert_bounded_depth(3)
    _assert_valid_phase("expanding")

    # ── Terminate (State 7): phase="complete" ─────────────────────────────────
    #    API call: extract_subgraph(NA) performs the full Seed→Expand→Terminate.
    result: SubgraphResult = chain_dag.extract_subgraph(NA)
    visited: set[str] = _node_ids(result)

    assert visited == {NA, NB, NC}, (
        f"Trace {trace_id}: expected visited={{A,B,C}}, got {visited!r}"
    )

    final_phase = "complete"
    final_depth = 3        # three edges traversed: A→B (d=1), B→C (d=2), C done (d=3)
    final_frontier: set[str] = set()

    _assert_all_invariants(visited, final_phase, final_depth, final_frontier)

def test_claudecode_removed_before_sdk_import(
    chain_dag: RegistryDag, monkeypatch: pytest.MonkeyPatch
) -> None:
    """
    GWT behaviour under test:
      Given: CLAUDECODE is set in os.environ.
      When:  extract_subgraph() (which internally calls _build_async_extract_fn())
             is about to import claude_agent_sdk.
      Then:  CLAUDECODE must be absent from os.environ at that moment.
    """
    monkeypatch.setenv("CLAUDECODE", "1")
    assert os.environ.get("CLAUDECODE") == "1"

    env_snapshot: dict[str, str | None] = {}
    _real_import = builtins.__import__

    def _capturing_import(name: str, *args, **kwargs):
        if name == "claude_agent_sdk":
            env_snapshot["CLAUDECODE"] = os.environ.get("CLAUDECODE", "__absent__")
        return _real_import(name, *args, **kwargs)

    with patch("builtins.__import__", side_effect=_capturing_import):
        try:
            chain_dag.extract_subgraph(NA)
        except Exception:
            # SDK may not be installed in the test environment;
            # the env-var state at import time is what we assert.
            pass

    if env_snapshot:
        assert env_snapshot.get("CLAUDECODE") == "__absent__", (
            "_build_async_extract_fn() must remove CLAUDECODE from os.environ "
            "before importing claude_agent_sdk; "
            f"observed CLAUDECODE={env_snapshot.get('CLAUDECODE')!r}"
        )


def test_claudecode_not_present_in_env_during_sdk_module_init(
    chain_dag: RegistryDag, monkeypatch: pytest.MonkeyPatch
) -> None:
    """
    Variant: verifies the removal also covers re-entrant import scenarios where
    claude_agent_sdk is already cached in sys.modules.
    Even if the module is cached, os.environ must not contain CLAUDECODE.
    """
    monkeypatch.setenv("CLAUDECODE", "claude-code-sentinel")

    env_during_call: list[str | None] = []

    _real_import = builtins.__import__

    def _spy_import(name: str, *args, **kwargs):
        result = _real_import(name, *args, **kwargs)
        if name == "claude_agent_sdk":
            env_during_call.append(os.environ.get("CLAUDECODE"))
        return result

    with patch("builtins.__import__", side_effect=_spy_import):
        try:
            chain_dag.extract_subgraph(NA)
        except Exception:
            pass

    for observed in env_during_call:
        assert observed is None, (
            "CLAUDECODE must be absent during claude_agent_sdk import; "
            f"got {observed!r}"
        )

        """
        ND = "D"
        dag = RegistryDag()
        for nid in (NA, NB, NC, ND):
            dag.add_node(Node.behavior(nid, nid, "g", "w", "t"))
        dag.add_edge(Edge(NA, NB, EdgeType.IMPORTS))
        dag.add_edge(Edge(NA, NC, EdgeType.IMPORTS))
        dag.add_edge(Edge(NB, ND, EdgeType.IMPORTS))
        dag.add_edge(Edge(NC, ND, EdgeType.IMPORTS))

        result = dag.extract_subgraph(NA)
        visited = _node_ids(result)

        assert {NA, NB, NC, ND}.issubset(visited)
        _assert_seed_always_included(visited, "complete")
        _assert_valid_phase("complete")
        _assert_bounded_depth(len(visited) - 1)   # ≤ 3, well within MaxSteps=4
        _assert_frontier_subset_visited(set(), visited)

    def test_seed_from_middle_of_chain(self) -> None:
        """
        Seeding from B in A→B→C: only {B, C} are reachable.
        SeedAlwaysIncluded requires B ∈ visited (start = B here).
        """
        dag = _make_chain_dag()
        result = dag.extract_subgraph(NB)
        visited = _node_ids(result)

        assert NB in visited
        assert NC in visited
        assert NA not in visited   # A is not reachable forward from B
        _assert_valid_phase("complete")
        _assert_bounded_depth(len(visited) - 1)

    def test_seed_from_leaf_node(self) -> None:
        """
        Seeding from C (leaf) in A→B→C: only {C} is reachable.
        frontier collapses to {} immediately after Seed.
        """
        dag = _make_chain_dag()
        result = dag.extract_subgraph(NC)
        visited = _node_ids(result)

        assert visited == {NC}
        _assert_frontier_subset_visited(set(), visited)
        _assert_bounded_depth(0)
        _assert_valid_phase("complete")

    def test_two_independent_chains_no_cross_contamination(self) -> None:
        """
        Two independent chains: A→B→C and D→E.
        Subgraph from A must not include D or E.
        """
        ND, NE = "D", "E"
        dag = RegistryDag()
        for nid in (NA, NB, NC, ND, NE):
            dag.add_node(Node.behavior(nid, nid, "g", "w", "t"))
        dag.add_edge(Edge(NA, NB, EdgeType.IMPORTS))
        dag.add_edge(Edge(NB, NC, EdgeType.IMPORTS))
        dag.add_edge(Edge(ND, NE, EdgeType.IMPORTS))

        result = dag.extract_subgraph(NA)
        visited = _node_ids(result)

        assert visited == {NA, NB, NC}
        assert ND not in visited
        assert NE not in visited
        _assert_all_invariants(visited, "complete", 3, set())

    def test_wide_fan_out_bounded_depth(self) -> None:
        """
        A node with four direct successors: A→{B,C,D,E}.
        All must be reachable at depth=1, well within MaxSteps=4.
        """
        children = [NB, NC, "D", "E"]
        dag = RegistryDag()
        dag.add_node(Node.behavior(NA, NA, "g", "w", "t"))
        for child in children:
            dag.add_node(Node.behavior(child, child, "g", "w", "t"))
            dag.add_edge(Edge(NA, child, EdgeType.IMPORTS))

        result = dag.extract_subgraph(NA)
        visited = _node_ids(result)

        expected = {NA} | set(children)
        assert expected.issubset(visited)
        _assert_bounded_depth(1)
        _assert_frontier_subset_visited(set(), visited)
        _assert_valid_phase("complete")

    def test_cycle_edge_rejected_or_traversal_still_terminates(self) -> None:
        """
        Self-loops / back-edges: either CycleError is raised at add_edge time
        (cycle detection), OR if accepted, extract_subgraph must still terminate.
        In either case BoundedDepth and ValidPhase must not be violated.
        """
        dag = RegistryDag()
        dag.add_node(Node.behavior(NA, NA, "g", "w", "t"))

        cycle_detected_at_add = False
        try:
            dag.add_edge(Edge(NA, NA, EdgeType.IMPORTS))
        except CycleError:
            cycle_detected_at_add = True

        if not cycle_detected_at_add:
            result = dag.extract_subgraph(NA)
            visited = _node_ids(result)
            assert NA in visited
            _assert_valid_phase("complete")
            _assert_bounded_depth(MAX_STEPS)   # bounded even in degenerate case

    def test_claudecode_set_does_not_alter_subgraph_result(
        self, monkeypatch: pytest.MonkeyPatch
    ) -> None:
        """
        Regression: CLAUDECODE presence must not change the topology returned.
        Both calls (with / without the env var) must produce identical visited sets.
        """
        dag_a = _make_chain_dag()
        result_without = _node_ids(dag_a.extract_subgraph(NA))

        monkeypatch.setenv("CLAUDECODE", "1")
        dag_b = _make_chain_dag()
        result_with = _node_ids(dag_b.extract_subgraph(NA))

        assert result_without == result_with == {NA, NB, NC}, (
            "Subgraph result must be identical regardless of CLAUDECODE; "
            f"without={result_without!r}, with={result_with!r}"
        )

    def test_all_invariants_hold_after_claudecode_removal(
        self, chain_dag: RegistryDag, monkeypatch: pytest.MonkeyPatch
    ) -> None:
        """
        Full invariant sweep when CLAUDECODE is present: the env-var removal
        must not perturb traversal correctness.
        """
        monkeypatch.setenv("CLAUDECODE", "1")
        result = chain_dag.extract_subgraph(NA)
        visited = _node_ids(result)
        _assert_all_invariants(visited, "complete", 3, set())

### Design rationale

| Section | What it covers |
|---|---|
| **Constants** | Mirror `NA/NB/NC`, `MaxSteps=4`, `REACHABLE`, `VALID_PHASES` verbatim from the TLA+ spec |
| **Invariant helpers** (`_assert_*`) | One function per TLA+ invariant; each can be called independently at every simulated state |
| **`_clean_claudecode_env` fixture** | `autouse` — guarantees env isolation between every test so CLAUDECODE state never bleeds |
| **`test_trace_forward_subgraph_chain`** | Parameterised over trace IDs 1–10; replays all seven TLA+ states symbolically, then calls `extract_subgraph(NA)` for the `Terminate` step and asserts State-7 final values |
| **CLAUDECODE tests** | Use `builtins.__import__` interception to capture `os.environ` at the exact moment `claude_agent_sdk` is imported; also test restore, idempotent absence, and result-parity |
| **`TestSeedAlwaysIncluded` … `TestValidPhase`** | Each verifies ≥ 2 distinct topologies; also tests the helper itself rejects violations |
| **`TestEdgeCases`** | Isolated nodes, empty DAG, `NodeNotFoundError`, diamond, mid-chain seed, leaf seed, two independent chains, wide fan-out, self-loop cycle handling, and two CLAUDECODE-specific regression tests |
