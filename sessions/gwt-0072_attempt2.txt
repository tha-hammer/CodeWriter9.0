MODULE TemplateSelectedByAnnotation

EXTENDS Integers, TLC, FiniteSets

CONSTANTS
    Templates,
    DefaultTemplate,
    ExistingFiles,
    UnknownTemplate,
    NoMetadata

ASSUME DefaultTemplate \in Templates
ASSUME ExistingFiles \subseteq Templates
ASSUME DefaultTemplate \in ExistingFiles
ASSUME UnknownTemplate \notin Templates
ASSUME NoMetadata \notin Templates
ASSUME NoMetadata # UnknownTemplate
