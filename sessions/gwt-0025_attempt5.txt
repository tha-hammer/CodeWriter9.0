---- MODULE ForwardSubgraphComplete ----

EXTENDS Integers, FiniteSets, TLC

CONSTANTS
    Nodes,
    StartNode,
    Succs,
    MaxDepth

RECURSIVE Reach(_, _)
Reach(S, boundary) ==
    IF boundary = {} THEN S
    ELSE
        LET next == ( UNION { IF n \in DOMAIN Succs THEN Succs[n] ELSE {}
                              : n \in boundary } ) \ S
        IN Reach(S \cup next, next)

Reachable == Reach({StartNode}, {StartNode})
