"""
Pytest test suite for StalenessPropagation.
Generated from 10 TLC-verified traces covering all seven TLA+ invariants:
  DirectlyStale, DirectCorrect, MonotonicGrowth, TransitiveComplete,
  NoFalseNegatives, AllDependentsIncluded, BoundedExecution.

import pytest
from dag import RegistryDag, NodeNotFoundError, CycleError
from registry.types import Edge, EdgeType, Node

# ─── constants ─────────────────────────────────────────────────────────────────
MAX_STEPS = 10

N1, N2 = "Nodes_1", "Nodes_2"
H1, H2 = "Hashes_1", "Hashes_2"

# ─── algorithm helpers (mirror TLA+ spec exactly) ──────────────────────────────

def _directly_stale(nodes, stored, current):
    """TLA+: DirectlyStale == {n ∈ Nodes : stored_hashes[n] ≠ current_hashes[n]}"""
    return frozenset(n for n in nodes if stored[n] != current[n])


def _one_hop_expand(stale, edges):
    """TLA+: OneHopExpand(S) == S ∪ {e[1] : e ∈ {ep ∈ edges : ep[2] ∈ S}}"""
    result = set(stale)
    for src, dst in edges:
        if dst in stale:
            result.add(src)
    return frozenset(result)


def _run_propagation(nodes, edges, stored, current, max_steps=MAX_STEPS):
    """
    Execute the TLA+ staleness-propagation algorithm and return final state.

    Mirrors:
      DetectDirect → PropLoop { PropStep → CheckFix } → Finish
    """
    # DetectDirect
    direct    = _directly_stale(nodes, stored, current)
    transitive = frozenset(direct)
    prev_size  = 0
    phase      = "propagate"
    step_count = 1

    # PropLoop / PropStep / CheckFix
    while phase == "propagate" and step_count < max_steps:
        prev_size  = len(transitive)                         # PropStep: snapshot
        transitive = _one_hop_expand(transitive, edges)      # PropStep: expand
        step_count += 1
        if len(transitive) == prev_size:                     # CheckFix
            phase = "complete"

    # Finish
    return dict(
        direct_stale    = direct,
        transitive_stale= transitive,
        phase           = phase,
        step_count      = step_count,
        prev_size       = prev_size,
    )


# ─── DAG construction ──────────────────────────────────────────────────────────

def _make_dag(node_ids, edge_pairs):
    """
    Build a RegistryDag from node IDs and (src, dst) edge pairs.
    Self-loops and cycle-inducing edges are silently skipped because
    RegistryDag enforces strict DAG constraints.
    """
    dag = RegistryDag()
    for nid in sorted(node_ids):
        dag.add_node(Node.behavior(nid, nid, "given", "when", "then"))
    for src, dst in sorted(edge_pairs):
        if src == dst:
            continue                              # skip self-loops
        try:
            dag.add_edge(Edge(src, dst, EdgeType.IMPORTS))
        except CycleError:
            pass                                  # skip cycle-inducing edges
    return dag


# ─── invariant verifier ────────────────────────────────────────────────────────

def _assert_invariants(nodes, edges, stored, current, result, max_steps=MAX_STEPS):
    """Assert all seven TLA+ invariants against a propagation result."""
    d  = result["direct_stale"]
    t  = result["transitive_stale"]
    ph = result["phase"]
    sc = result["step_count"]
    ps = result["prev_size"]

    # DirectCorrect: after DetectDirect, direct_stale == DirectlyStale
    assert d == _directly_stale(nodes, stored, current), \
        f"DirectCorrect violated: direct_stale={d}"

    # MonotonicGrowth: |transitive_stale| >= prev_size
    assert len(t) >= ps, \
        f"MonotonicGrowth violated: |t|={len(t)} < prev_size={ps}"

    # BoundedExecution: step_count <= MaxSteps
    assert sc <= max_steps, \
        f"BoundedExecution violated: step_count={sc} > max_steps={max_steps}"

    if ph == "complete":
        # TransitiveComplete – fixed point
        expanded = _one_hop_expand(t, edges)
        assert expanded == t, \
            f"TransitiveComplete(fixed-point) violated: expanded={expanded}, t={t}"

        # TransitiveComplete – direct_stale ⊆ transitive_stale
        assert d <= t, \
            f"TransitiveComplete(subset) violated: direct={d} not ⊆ transitive={t}"

        # NoFalseNegatives: ∀ n ∈ direct_stale ⟹ n ∈ transitive_stale
        for n in nodes:
            if n in d:
                assert n in t, \
                    f"NoFalseNegatives violated: {n} in direct_stale but not transitive_stale"

        # AllDependentsIncluded: ∀ (src,dst) ∈ edges, dst ∈ t ⟹ src ∈ t
        for src, dst in edges:
            if dst in t:
                assert src in t, \
                    f"AllDependentsIncluded violated: edge ({src},{dst}), {dst} stale but {src} not"

Pytest test suite for StalenessPropagation.
Generated from 10 TLC-verified traces covering all seven TLA+ invariants:
  DirectlyStale, DirectCorrect, MonotonicGrowth, TransitiveComplete,
  NoFalseNegatives, AllDependentsIncluded, BoundedExecution.

import pytest
from dag import RegistryDag, NodeNotFoundError, CycleError
from registry.types import Edge, EdgeType, Node

# ─── constants ─────────────────────────────────────────────────────────────────
MAX_STEPS = 10

N1, N2 = "Nodes_1", "Nodes_2"
H1, H2 = "Hashes_1", "Hashes_2"

# ─── algorithm helpers (mirror TLA+ spec exactly) ──────────────────────────────

def _directly_stale(nodes, stored, current):
    """TLA+: DirectlyStale == {n ∈ Nodes : stored_hashes[n] ≠ current_hashes[n]}"""
    return frozenset(n for n in nodes if stored[n] != current[n])


def _one_hop_expand(stale, edges):
    """TLA+: OneHopExpand(S) == S ∪ {e[1] : e ∈ {ep ∈ edges : ep[2] ∈ S}}"""
    result = set(stale)
    for src, dst in edges:
        if dst in stale:
            result.add(src)
    return frozenset(result)


def _run_propagation(nodes, edges, stored, current, max_steps=MAX_STEPS):
    """
    Execute the TLA+ staleness-propagation algorithm and return final state.

    Mirrors:
      DetectDirect → PropLoop { PropStep → CheckFix } → Finish
    """
    # DetectDirect
    direct    = _directly_stale(nodes, stored, current)
    transitive = frozenset(direct)
    prev_size  = 0
    phase      = "propagate"
    step_count = 1

    # PropLoop / PropStep / CheckFix
    while phase == "propagate" and step_count < max_steps:
        prev_size  = len(transitive)                         # PropStep: snapshot
        transitive = _one_hop_expand(transitive, edges)      # PropStep: expand
        step_count += 1
        if len(transitive) == prev_size:                     # CheckFix
            phase = "complete"

    # Finish
    return dict(
        direct_stale    = direct,
        transitive_stale= transitive,
        phase           = phase,
        step_count      = step_count,
        prev_size       = prev_size,
    )


# ─── DAG construction ──────────────────────────────────────────────────────────

def _make_dag(node_ids, edge_pairs):
    """
    Build a RegistryDag from node IDs and (src, dst) edge pairs.
    Self-loops and cycle-inducing edges are silently skipped because
    RegistryDag enforces strict DAG constraints.
    """
    dag = RegistryDag()
    for nid in sorted(node_ids):
        dag.add_node(Node.behavior(nid, nid, "given", "when", "then"))
    for src, dst in sorted(edge_pairs):
        if src == dst:
            continue                              # skip self-loops
        try:
            dag.add_edge(Edge(src, dst, EdgeType.IMPORTS))
        except CycleError:
            pass                                  # skip cycle-inducing edges
    return dag


# ─── invariant verifier ────────────────────────────────────────────────────────

def _assert_invariants(nodes, edges, stored, current, result, max_steps=MAX_STEPS):
    """Assert all seven TLA+ invariants against a propagation result."""
    d  = result["direct_stale"]
    t  = result["transitive_stale"]
    ph = result["phase"]
    sc = result["step_count"]
    ps = result["prev_size"]

    # DirectCorrect: after DetectDirect, direct_stale == DirectlyStale
    assert d == _directly_stale(nodes, stored, current), \
        f"DirectCorrect violated: direct_stale={d}"

    # MonotonicGrowth: |transitive_stale| >= prev_size
    assert len(t) >= ps, \
        f"MonotonicGrowth violated: |t|={len(t)} < prev_size={ps}"

    # BoundedExecution: step_count <= MaxSteps
    assert sc <= max_steps, \
        f"BoundedExecution violated: step_count={sc} > max_steps={max_steps}"

    if ph == "complete":
        # TransitiveComplete – fixed point
        expanded = _one_hop_expand(t, edges)
        assert expanded == t, \
            f"TransitiveComplete(fixed-point) violated: expanded={expanded}, t={t}"

        # TransitiveComplete – direct_stale ⊆ transitive_stale
        assert d <= t, \
            f"TransitiveComplete(subset) violated: direct={d} not ⊆ transitive={t}"

        # NoFalseNegatives: ∀ n ∈ direct_stale ⟹ n ∈ transitive_stale
        for n in nodes:
            if n in d:
                assert n in t, \
                    f"NoFalseNegatives violated: {n} in direct_stale but not transitive_stale"

        # AllDependentsIncluded: ∀ (src,dst) ∈ edges, dst ∈ t ⟹ src ∈ t
        for src, dst in edges:
            if dst in t:
                assert src in t, \
                    f"AllDependentsIncluded violated: edge ({src},{dst}), {dst} stale but {src} not"
