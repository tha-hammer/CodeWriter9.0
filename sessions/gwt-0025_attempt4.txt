EXTENDS Integers, FiniteSets, TLC

CONSTANTS
    Nodes,
    StartNode,
    Edges,
    MaxDepth

RECURSIVE Reach(_, _)
Reach(S, boundary) ==
    IF boundary = {} THEN S
    ELSE
        LET next == { e[2] : e \in { ed \in Edges :
                                      ed[1] \in boundary /\
                                      ed[2] \in Nodes } } \ S
        IN Reach(S \cup next, next)

Reachable == Reach({StartNode}, {StartNode})
