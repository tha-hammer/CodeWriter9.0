---- MODULE ValidateDependsOn ----

EXTENDS Integers, FiniteSets, TLC

CONSTANTS
    AllUUIDs,
    DependsOnInput,
    MaxSteps

(* --algorithm ValidateDependsOn

variables
    remaining         = DependsOnInput,
    depends_on_output = {},
    warnings          = 0,
    exception_raised  = FALSE,
    step_count        = 0;

define

    OutputSubsetOfAll ==
        depends_on_output \subseteq AllUUIDs

    NoExceptionRaised ==
        exception_raised = FALSE

    BoundedExecution ==
        step_count <= MaxSteps

    OutputConsistency ==
        depends_on_output = (DependsOnInput \ remaining) \cap AllUUIDs

    WarningsConsistency ==
        warnings = Cardinality((DependsOnInput \ remaining) \ AllUUIDs)

    NoInvalidSurvives ==
        depends_on_output \cap (DependsOnInput \ AllUUIDs) = {}

    TerminalCondition ==
        (remaining = {}) => (depends_on_output \subseteq AllUUIDs)

    AllValidRetained ==
        (remaining = {}) =>
            (DependsOnInput \cap AllUUIDs \subseteq depends_on_output)

    OutputIsExactIntersection ==
        (remaining = {}) =>
            (depends_on_output = DependsOnInput \cap AllUUIDs)

end define;

fair process validator = "validator"
begin
    Validate:
        while remaining # {} /\ step_count < MaxSteps do
            with uuid \in remaining do
                remaining := remaining \ {uuid};
                if uuid \in AllUUIDs then
                    depends_on_output := depends_on_output \cup {uuid};
                else
                    warnings := warnings + 1;
                end if;
            end with;
            step_count := step_count + 1;
        end while;
    Finish:
        skip;
end process;

end algorithm; *)
